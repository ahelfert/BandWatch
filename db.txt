
CREATE TABLE `venue` (
	`venue_id`	INTEGER,
	`name`	TEXT NOT NULL,
	`location`	TEXT,
	`city`	TEXT NOT NULL,
	`country`	TEXT,
	`region`	TEXT,
	PRIMARY KEY(`venue_id`)
);
CREATE TABLE `event_artist` (
	`event_id`	INTEGER NOT NULL,
	`arid`	INTEGER NOT NULL,
	PRIMARY KEY(`event_id`,`arid`),
	FOREIGN KEY(`event_id`) REFERENCES `event`(`event_id`),
	FOREIGN KEY(`arid`) REFERENCES `artist`(`arid`)
);
CREATE TABLE `event` (
	`event_id`	INTEGER,
	`title`	TEXT,
	`date`	TEXT NOT NULL,
	`description`	TEXT,
	`venue_id`	INTEGER NOT NULL,
	PRIMARY KEY(`event_id`),
	FOREIGN KEY(`venue_id`) REFERENCES `venue`(`venue_id`)
);
CREATE TABLE `artist` (
	`arid`	INTEGER,
	`name`	TEXT NOT NULL,
	PRIMARY KEY(`arid`)
);
CREATE TABLE `album` (
	`reid`	INTEGER,
	`title`	TEXT NOT NULL,
	`type`	TEXT,
	`date`	TEXT NOT NULL,
	`arid`	INTEGER NOT NULL,
	`rgid`	INTEGER,
	PRIMARY KEY(`reid`),
	FOREIGN KEY(`arid`) REFERENCES artist(arid)
);
